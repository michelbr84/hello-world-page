Hello, World! in Programming Languages

This page shows how to print "Hello, World!" in several popular programming languages.

Python

print("Hello, World!")

JavaScript

console.log("Hello, World!");

Java

public class HelloWorld {
    public static void main(String[] args) {
        System.out.println("Hello, World!");
    }
}

C

#include 
int main() {
    printf("Hello, World!\n");
    return 0;
}

C++

#include 
int main() {
    std::cout << "Hello, World!" << std::endl;
    return 0;
}

C#

using System;
class Program {
    static void Main() {
        Console.WriteLine("Hello, World!");
    }
}

Go

package main
import "fmt"
func main() {
    fmt.Println("Hello, World!")
}

Ruby

puts "Hello, World!"

PHP

Swift

print("Hello, World!")

Haskell

main = putStrLn "Hello, World!"

Perl

print "Hello, World!\n";

Lua

print("Hello, World!")

R

cat("Hello, World!\n")

Objective-C

#import <stdio.h>
int main() {
    printf("Hello, World!\n");
    return 0;
}

Rust

fn main() {
    println!("Hello, World!");
}

Kotlin

fun main() {
    println("Hello, World!")
}

TypeScript

console.log("Hello, World!");
